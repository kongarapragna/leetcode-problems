class Solution {
public:
    bool isPalindrome(int x) {
        int temp=x;
        long long rev=0;
        int ld;
        if(x<0){
            return false;
        }
        while(x){
            ld=x%10;
            rev=rev*10+ld;
            x/=10;
        }
        if(rev==temp){
            return true;
        }
        return false;
    }
};
